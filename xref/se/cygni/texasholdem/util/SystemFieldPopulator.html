23      public void populateValuesFromSystemProperties() {
24  
25          final Field[] fields = target.getClass().getDeclaredFields();
26          for (final Field f : fields) {
27  
28              // Don't try to set final variables
29              if (Modifier.isFinal(f.getModifiers()))
30                  continue;
31  
32              final Object valueFromSystem = getPropertyFromSystem(
33                      f.getName(), f.getType());
34  
35              if (valueFromSystem != null) {
36                  f.setAccessible(true);
37                  try {
38                      f.set(target, valueFromSystem);
39  
40                      log.info("Overriding property {} to {} for "
41                              + target.getClass().getSimpleName(), f.getName(),
42                              valueFromSystem);
43  
44                  } catch (final Exception e) {
45                      log.warn("Failed to override property {} to {} for "
46                              + target.getClass().getSimpleName(), f.getName(),
47                              valueFromSystem);
48                  }
49              }
50          }
51      }

54      private Object getPropertyFromSystem(final String property, final Class type) {
55  
56          if (type == Double.class || type == double.class) {
57              return getDoublePropertyFromSystem(property);
58          }
59  
60          if (type == Float.class || type == float.class) {
61              return getFloatPropertyFromSystem(property);
62          }
63  
64          if (type == Long.class || type == long.class) {
65              return getLongPropertyFromSystem(property);
66          }
67  
68          if (type == Integer.class || type == int.class) {
69              return getIntegerPropertyFromSystem(property);
70          }
71  
72          if (type == String.class) {
73              return getStringPropertyFromSystem(property);
74          }
75

77              return getBooleanPropertyFromSystem(property);
78          }

80          return null;
